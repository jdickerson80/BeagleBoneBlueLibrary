39 IOBase::IOBase( int number, Direction::Enum whatDirection )
40  : _threadRunning( false )
41  , _number( number )
42 {
43  // set up the name and path variables
44  ostringstream s;
45  s << "gpio" << this->_number;
46  this->_name = string(s.str());
47  this->_path = GPIO_PATH + this->_name + "/";
48 
49  // call export on the pin to created it
50  this->exportGPIO();
51 
52  // wait for Linux to set up the sysfs structure
53  usleep( 250000 ); // 250ms delay
54 
55  // set the direction of the IO
56  setDirection( whatDirection );
57 }
